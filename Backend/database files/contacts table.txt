CREATE TABLE contacts (
    sl_no SERIAL,
    name VARCHAR(500),
    customer_unique_code VARCHAR(500),
    clinic_college_name VARCHAR(500),
    designation VARCHAR(500),
    department VARCHAR(500),
    address VARCHAR(500),
    mobileno BIGINT PRIMARY KEY,
    whatsapp_availability BOOLEAN DEFAULT FALSE,
    alternative_mobile_no BIGINT,
    alternative_mobile_no2 BIGINT,
    alternative_mobile_no3 BIGINT,
    telephone VARCHAR(500),
    drug_license_no VARCHAR(50),
    gst VARCHAR(15),
    email_id VARCHAR(500),
    website VARCHAR(500),
    city VARCHAR(500),
    state VARCHAR(500),
    country VARCHAR(500),
    district VARCHAR(500),
    pincode CHAR(500),
    type VARCHAR(500),
    source VARCHAR(500),
    status VARCHAR(500),
    enquiry VARCHAR(500),
    last_purchased_date DATE,
    branch_data VARCHAR(500),
    under_sales_person VARCHAR(500),
    create_date DATE,
    age VARCHAR(500),
    tags VARCHAR(500),
    last_updated_date DATE,
    -- Constraints
    CONSTRAINT mobile_no_check CHECK (mobileno >= 1000000000 AND mobileno <= 9999999999), -- Checking for 10-digit mobile numbers directly
    CONSTRAINT alt_mobile_no1_check CHECK (alternative_mobile_no IS NULL OR (alternative_mobile_no >= 1000000000 AND alternative_mobile_no <= 9999999999)),
    CONSTRAINT alt_mobile_no2_check CHECK (alternative_mobile_no2 IS NULL OR (alternative_mobile_no2 >= 1000000000 AND alternative_mobile_no2 <= 9999999999)),
    CONSTRAINT alt_mobile_no3_check CHECK (alternative_mobile_no3 IS NULL OR (alternative_mobile_no3 >= 1000000000 AND alternative_mobile_no3 <= 9999999999)),
    CONSTRAINT create_date_check CHECK (create_date IS NULL OR create_date = date_trunc('day', create_date)),
    CONSTRAINT last_updated_date_check CHECK (last_updated_date IS NULL OR last_updated_date = date_trunc('day', last_updated_date))
);
